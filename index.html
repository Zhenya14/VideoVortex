// Load videos and update views on click function loadVideos() { const videoGallery = document.getElementById("video-gallery"); videoGallery.innerHTML = ""; database.ref("videos").once("value").then((snapshot) => { snapshot.forEach((childSnapshot) => { const videoData = childSnapshot.val(); const videoKey = childSnapshot.key; // Key for updating view count const videoElement = document.createElement("video"); videoElement.src = videoData.url; videoElement.controls = true; videoElement.classList.add("video-item"); videoElement.onclick = () => { updateViews(videoKey, videoData.views + 1); // Update views on click }; const infoElement = document.createElement("div"); infoElement.classList.add("video-info"); // Avatar const avatar = document.createElement("div"); avatar.classList.add("avatar"); avatar.style.backgroundColor = getRandomColor(); const initials = videoData.author ? videoData.author.charAt(0).toUpperCase() : "?"; avatar.textContent = initials; // Video details const videoDetails = document.createElement("div"); videoDetails.classList.add("video-details"); const date = new Date(videoData.date).toLocaleDateString(); videoDetails.innerHTML = ` ${videoData.author}
${videoData.title} (${videoData.views} переглядів)
${date} `; // Append avatar and details infoElement.appendChild(avatar); infoElement.appendChild(videoDetails); videoGallery.appendChild(infoElement); videoGallery.appendChild(videoElement); }); }); } // Update views without authentication check function updateViews(videoKey, views) { database.ref("videos/" + videoKey).update({ views: views }); } // Load videos on page load window.onload = function() { loadVideos(); };
